    # File api/Timer.rb, line 55
def tick
        @elapsed = Time.now - @start
        time=@accumulated+@elapsed
        h = sprintf('%02i', (time.to_i / 3600))
        m = sprintf('%02i', ((time.to_i % 3600) / 60))
        s = sprintf('%02i', (time.to_i % 60))
        newtime = "#{h}:#{m}:#{s}"
end
