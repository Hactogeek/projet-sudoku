def redo
        if self.canRedo? then
                dernierMemento = Marshal.dump(@partie.getPlateau())
                @undos.push(dernierMemento)
                plateau = @redos.pop
                plateau = Marshal.load(plateau)
                @partie.setPlateau(plateau)
                print("\n","Redo effectue")
        end
        return nil
end
